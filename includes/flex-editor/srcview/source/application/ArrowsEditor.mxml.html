    <fx:Script>
        <![CDATA[
            import data.GridBuilder;
            
            import enums.ArrowType;
            import enums.DefinitionOrientation;
            import enums.SquareType;
            import enums.TypingDirection;
            
            import model.Definition;
            import model.Grid;
            import model.Square;
            
            import mx.collections.ArrayCollection;
            import mx.collections.IList;
            import mx.collections.Sort;
            import mx.collections.SortField;
            import mx.events.FlexEvent;
            
            import renderers.EditorDefinitionItemRenderer;
            
            import spark.events.IndexChangeEvent;
            
            private var _grid : Grid;
            
            private function creationCompleteHandler(event:FlexEvent):void{
                var builder : GridBuilder = new GridBuilder();
                _grid = builder.createBlankGrid(7,5);
                (gridUI.layout as TileLayout).requestedColumnCount = _grid.colCount;
                (gridUI.layout as TileLayout).requestedRowCount = _grid.rowCount;
                gridUI.dataProvider = new ArrayCollection(_grid.squares);
                gridUI.setFocus();
            }
            
            private function gridUI_keyUpHandler(event:KeyboardEvent):void{
                var selectedSquare : Square = gridUI.selectedItem as Square;
                
                if (selectedSquare.solved) return;
                
                var num : int = event.keyCode;
                
                if (num > 64 && num < 91){
                    var strCaps:String = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
                    var char : String = strCaps.charAt(num - 65);
                    
                    selectedSquare.type = SquareType.LETTER;
                    selectedSquare.letter = char;
                    
                    (gridUI.dataProvider as ArrayCollection).itemUpdated(selectedSquare);
                }else if(event.keyCode == Keyboard.SPACE){
                    selectedSquare.letter = "";
                    if (selectedSquare.type == SquareType.LETTER){
                        selectedSquare.type = SquareType.DEFINITION;
                    }else{
                        selectedSquare.type = SquareType.LETTER;                        
                    }
                    (gridUI.dataProvider as ArrayCollection).itemUpdated(selectedSquare);
                }else if(event.keyCode == Keyboard.DELETE || event.keyCode == Keyboard.BACKSPACE){
                    selectedSquare.type = SquareType.LETTER;
                    selectedSquare.letter = "";
                    (gridUI.dataProvider as ArrayCollection).itemUpdated(selectedSquare);
                }
                
                var horWords : ArrayCollection = new ArrayCollection(_grid.getHorizontalWords());
                var verWords : ArrayCollection = new ArrayCollection(_grid.getVerticalWords());
                
                var defSort : Sort = new Sort();
                defSort.fields = [new SortField("startIndex")];
                
                horWords.sort = verWords.sort = defSort;
                horWords.refresh();
                verWords.refresh();
                
                horWordList.dataProvider = horWords;
                verWordList.dataProvider = verWords;
            }
            
            private function goXMLBtn_clickHandler(event:MouseEvent):void {
                xmlTA.visible = xmlTA.includeInLayout = true;
                                
                var xml : XML = _grid.getXML();
                
                xmlTA.text = xml.toXMLString();
            }

        ]]>
    </fx:Script>
