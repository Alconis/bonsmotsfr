package data
{
    import enums.ArrowType;
    import enums.DefinitionOrientation;
    import enums.SquareType;
    
    import model.Definition;
    import model.Grid;
    import model.Square;

    public class GridBuilder
    {
        public static const NS : Namespace = new Namespace("","alco");
                
        private var _parsedGrid : Grid;
        
        public function get parsedGrid() : Grid {
            return _parsedGrid;
        }
        
        public function createBlankGrid( rowCount : int, colCount : int ) : Grid {
            var g : Grid = new Grid();
            
            g.rowCount = rowCount;
            g.colCount = colCount;
            
            var i : int=0;
            var end : int = g.rowCount * g.colCount;
            var square : Square;
            while(i < end){
                square = new Square();
                square.idx = i;
                square.gridRef = g;
                
                // First line: def for all odd index
                if ((i < g.colCount) && (i % 2) == 0){
                    square.type = SquareType.DEFINITION;
                    square.solved = true;
                // First column: def for all colCount*2
                }else if(i % (g.colCount * 2) == 0){
                    square.type = SquareType.DEFINITION;
                    square.solved = true;
                }else{
                    square.type = SquareType.LETTER;
                    square.solved = false;
                }
                
                g.squares.push(square);
                i++;
            }
            
            return g;
        } 
        
        private var _xml : XML;
        public function parseXML( xml : XML ) : Boolean {
            var result : Boolean= false;
            
            _xml = xml;
            
            _parsedGrid = new Grid();
                        
            _parsedGrid.rowCount = int(_xml.@rowCount);
            _parsedGrid.colCount = int(_xml.@colCount);
            
            var i : int=0;
            var end : int = _parsedGrid.rowCount * _parsedGrid.colCount;
            
            // Build all squares setting type (letter or def)
            var words : XMLList = _xml.word;
            var wordsAtCurrentIdx : XMLList;
            var square : Square;
            while(i < end){
                square = new Square();
                square.idx = i;
                square.gridRef = _parsedGrid;
                
                wordsAtCurrentIdx = words.(@idx==i);
                if (0 == wordsAtCurrentIdx.length()){
                    square.type = SquareType.LETTER;
                }else{
                    square.type = SquareType.DEFINITION;
                    
                    var xmlDef : XML;
                    var definition : Definition;
                    for each(xmlDef in wordsAtCurrentIdx){
                        definition = new Definition();
                        definition.text = xmlDef.@definition;
                        definition.orientation = xmlDef.@orientation;
                        definition.solution = xmlDef.@solution
                            
                        square.definitions.push(definition);
                    }
                }
                _parsedGrid.squares.push(square);
                i++;
            }
            
            // Re run though the grid to add arrows information.
            i = 0;
            var targetSq : Square;
            while(i < end){
                square = _parsedGrid.squares[i];
                
                if (square.type == SquareType.DEFINITION){
                    for each(definition in square.definitions){
                        if (definition.orientation == DefinitionOrientation.HORIZONTAL){
                            targetSq = _parsedGrid.squares[i+1];
                            if(square.definitions.length > 1){
                                targetSq.arrows.push(ArrowType.HOR_TOP);
                            }else{
                                targetSq.arrows.push(ArrowType.HOR);
                            }
                            continue;
                        }
                        if (definition.orientation == DefinitionOrientation.HORIZONTAL_UNDER){
                            targetSq = _parsedGrid.squares[i+_parsedGrid.colCount];
                            targetSq.arrows.push(ArrowType.HOR_UNDER);
                            continue;
                        }
                        if (definition.orientation == DefinitionOrientation.VERTICAL){
                            targetSq = _parsedGrid.squares[i+_parsedGrid.colCount];
                            targetSq.arrows.push(ArrowType.VER);
                            continue;
                        }
                        if (definition.orientation == DefinitionOrientation.VERTICAL_RIGHT){
                            targetSq = _parsedGrid.squares[i+1];
                            if(square.definitions.length > 1){
                                targetSq.arrows.push(ArrowType.VER_RIGHT_TOP);
                            }else{
                                targetSq.arrows.push(ArrowType.VER_RIGHT);
                            }
                            continue;
                        }
                    }
                }
                
                i++;
            }
            
            // Re run through the definitions to set the solution letters on every LETTER square
            var defIdx : int;
            var defOrientation : String;
            var defSolution : String;
            var startIdx : int;
            var endIdx : int;
            var idxIncr : int;
            for each(xmlDef in words){
                defIdx = xmlDef.@idx;
                defOrientation = xmlDef.@orientation;
                defSolution = xmlDef.@solution;
                
                if (defOrientation == DefinitionOrientation.HORIZONTAL){
                    idxIncr = 1;
                    startIdx = defIdx + 1;
                }else if (defOrientation == DefinitionOrientation.HORIZONTAL_UNDER){
                    idxIncr = 1;
                    startIdx = defIdx + _parsedGrid.colCount;
                }else if (defOrientation == DefinitionOrientation.VERTICAL){
                    idxIncr = _parsedGrid.colCount;
                    startIdx = defIdx + _parsedGrid.colCount;
                }else if (defOrientation == DefinitionOrientation.VERTICAL_RIGHT){
                    idxIncr = _parsedGrid.colCount;
                    startIdx = defIdx + 1;
                }else{
                    continue;
                }
                
                var j : int = 0;
                endIdx = startIdx + ((defSolution.length-1) * idxIncr);
                for(i = startIdx; i <= endIdx; i += idxIncr){
                    square = _parsedGrid.squares[i] as Square;
                    square.solution = defSolution.charAt(j++);
                }
            }
            
            // Do some validation check
            result = true;
            
            return result;
        }
    }
}
