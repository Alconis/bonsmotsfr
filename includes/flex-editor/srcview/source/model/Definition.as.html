package model
{
    import enums.DefinitionOrientation;

    public class Definition
    {
        public var startIndex : int=-1;
        public var defIndex : int=-1;
        public var text : String="";
        public var orientation : String = DefinitionOrientation.NONE;
        public var solution : String="";
        
        public function toXML() : XML {
            return <word idx={defIndex} definition={text} orientation={orientation} solution={solution}/>
        }
    }
}
