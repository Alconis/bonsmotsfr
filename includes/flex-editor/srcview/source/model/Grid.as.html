package model
{
    import enums.DefinitionOrientation;
    import enums.SquareType;
    import enums.TypingDirection;
    
    import flash.utils.Dictionary;
    
    import mx.collections.ArrayCollection;
    import mx.collections.Sort;
    import mx.collections.SortField;

    public class Grid
    {        
        public var rowCount : int;
        public var colCount : int;
        
        public var typingDirection : String;
        public var showSolution : Boolean;
        
        public function Grid(){
            typingDirection = TypingDirection.RIGHT;
            showSolution = false;
        }
        
        private var _squares : Array=[];
        
        public function get squares() : Array {
            return _squares;
        }
        
        public function set squares( value : Array ) : void {
            _squares = value;
        }
        
        /*
         * Functions mainly used by the editor
         */
        
        public function getXML() : XML {
            var horWords : ArrayCollection = new ArrayCollection(getHorizontalWords());
            var verWords : ArrayCollection = new ArrayCollection(getVerticalWords());
            
            var allWords : ArrayCollection = new ArrayCollection();
            allWords.addAll(horWords);
            allWords.addAll(verWords);
            
            var def : Definition;
            // Update definition orientations and definition indexes
            for(var i : int=0; i < allWords.length; i++){
                def = allWords.getItemAt(i) as Definition;
                updateDefinitionIdx(def);
            }
            
            // Add "none" definition
            var noneDefIdx : Array = new Array();
            for each(var sq : Square in _squares){
                if (sq.type == SquareType.DEFINITION){
                    var find : Boolean = false;
                    for each(def in allWords){
                        if (def.defIndex == sq.idx){
                            find = true;
                            break;
                        }
                    }
                    if (!find){
                        def = new Definition();
                        def.defIndex = sq.idx;
                        def.orientation = DefinitionOrientation.NONE;
                        def.solution = "";
                        def.text = "";
                        allWords.addItem(def);
                    }
                }
            }
            
            // Sort by defIndex
            var defSort : Sort = new Sort();
            defSort.fields = [new SortField("startIndex")];
            allWords.sort = defSort;
            allWords.refresh();
            
            // Generate XML
            var xml : XML = <grid colCount={colCount} rowCount={rowCount}/>;
            for(i = 0; i < allWords.length; i++){
                def = allWords.getItemAt(i) as Definition;
                xml.appendChild(def.toXML());
            }
            
            return xml;
        }
        
        public function getHorizontalWords() : Array {
            var words : Array = new Array();
            var idx : int = 0;
            var end : int = colCount * rowCount;
            var word : String;
            var def : Definition;
            while(idx < end){
                word = getHorizontalWordAt(idx);
                if(word != "" && word.length > 1){
                    def = new Definition();
                    def.orientation = DefinitionOrientation.HORIZONTAL;
                    def.solution = word;
                    def.startIndex = idx;
                    words.push(def);
                    idx += word.length;
                }else{
                    idx++;
                }
            }

            return words;
        }
        
        public function getVerticalWords() : Array {
            var words : Array = new Array();
            var idx : int;
            var end : int = colCount * rowCount;
            var word : String;
            var def : Definition;
            for(var colIdx : int=0; colIdx < colCount; colIdx++){
                idx = colIdx;
                while(idx < end){
                    word = getVerticalWordAt(idx);
                    if(word != "" && word.length > 1){
                        def = new Definition();
                        def.orientation = DefinitionOrientation.VERTICAL;
                        def.solution = word;
                        def.startIndex = idx;
                        words.push(def);
                        idx += (word.length) * colCount;
                    }else{
                        idx += colCount;
                    }
                }
            }
            return words;
        }
        
        public function updateDefinitionIdx( def : Definition ) : void {
            if(def.orientation == DefinitionOrientation.HORIZONTAL){
                if (def.startIndex % colCount == 0){
                    def.defIndex = def.startIndex - colCount;
                    def.orientation = DefinitionOrientation.HORIZONTAL_UNDER;
                }else{
                    def.defIndex = def.startIndex - 1;
                }
            }else if(def.orientation == DefinitionOrientation.VERTICAL){
                if (def.startIndex < colCount){
                    def.defIndex = def.startIndex - 1;
                    def.orientation = DefinitionOrientation.VERTICAL_RIGHT;
                }else{
                    def.defIndex = def.startIndex - colCount;
                }                
            }
        }
        
        public function getHorizontalWordAt( idx : int ) : String {
            var square : Square = _squares[idx] as Square;
            if (square.type == SquareType.DEFINITION) return "";
            
            var word : String = square.letter == "" ? "-" : square.letter;
            var i : int = idx-1;
            var j : int = idx+1;
            var start : int = idx - (idx % colCount);
            var end : int = start + colCount - 1;
            
            var bSquare : Square;
            var aSquare : Square;
            var letter : String;
            while(i >= start || j <= end){
                bSquare = _squares[i] as Square;
                aSquare = _squares[j] as Square;
                
                if(i >= start && bSquare.type == SquareType.LETTER){
                    letter = bSquare.letter == "" ? "-" : bSquare.letter;
                    word = letter + word;
                    i--;
                }else{
                    i = start - 1;
                }
            
                if(j <= end && aSquare.type == SquareType.LETTER){
                    letter = aSquare.letter == "" ? "-" : aSquare.letter;
                    word = word + letter;
                    j++;
                }else{
                    j = end + 1;
                }
                
            }
            return word;
        }
        
        public function getVerticalWordAt( idx : int ) : String {
            var square : Square = _squares[idx] as Square;
            if (square.type == SquareType.DEFINITION) return "";
            
            var word : String = square.letter == "" ? "-" : square.letter;
            var i : int = idx-colCount;
            var j : int = idx+colCount;
            var start : int = 0;
            var end : int = colCount * rowCount;
            
            var bSquare : Square;
            var aSquare : Square;
            var letter : String;
            while(i >= start || j <= end){
                bSquare = _squares[i] as Square;
                aSquare = _squares[j] as Square;
                
                if(i >= start && bSquare.type == SquareType.LETTER){
                    letter = bSquare.letter == "" ? "-" : bSquare.letter;
                    word = letter + word;
                    i -= colCount;
                }else{
                    i = start - 1;
                }
                
                if(j <= end && aSquare.type == SquareType.LETTER){
                    letter = aSquare.letter == "" ? "-" : aSquare.letter;
                    word = word + letter;
                    j += colCount;
                }else{
                    j = end + 1;
                }
                
            }
            return word;
        }
    }
}
