    <fx:Script>
        <![CDATA[
            import components.EndGameWindow;
            
            import data.GridBuilder;
            
            import enums.ArrowType;
            import enums.SquareType;
            import enums.TypingDirection;
            
            import model.Grid;
            import model.Square;
            
            import mx.collections.ArrayCollection;
            import mx.collections.ICollectionView;
            import mx.controls.Alert;
            import mx.core.Application;
            import mx.core.FlexGlobals;
            import mx.events.FlexEvent;
            import mx.managers.BrowserManager;
            import mx.managers.IBrowserManager;
            import mx.managers.PopUpManager;
            import mx.rpc.events.FaultEvent;
            import mx.rpc.events.ResultEvent;
            import mx.utils.Base64Decoder;
            import mx.utils.URLUtil;
            
            import renderers.DefinitionItemRenderer;
            import renderers.LetterItemRenderer;
            
            import spark.effects.easing.Elastic;
            import spark.events.IndexChangeEvent;
            import spark.layouts.HorizontalLayout;
            
            [Bindable]private var _hideTools : Boolean=true;
            [Bindable]private var _inCorrection : Boolean=false;
            private var _requestedGridId : int;
            private var _grid : Grid;
            
            private function creationCompleteHandler(event:FlexEvent):void{
                var gridId : String;
                
                // Retrieve id from flashvars
                gridId = (FlexGlobals.topLevelApplication as Arrows).parameters.gridId;
                
                // Retrieve id from url
                if (!gridId){
                    var url : String = ExternalInterface.call("window.location.search.toString");
                    if (url){
                        var params : Object = URLUtil.stringToObject(url.substr(1),'&');
                        gridId = params.gridId;
                    }    
                }
                
                if (!gridId){
                    gridId = "5";
                }
                _requestedGridId = int(gridId);
                
                var gridDataBase64 : String = (FlexGlobals.topLevelApplication as Arrows).parameters.gridXML;
                
                if (!gridDataBase64 || gridDataBase64 == ""){
                    getGridXMLById.send({id : gridId});
                }else{
                    var resultEvt : ResultEvent = new ResultEvent(ResultEvent.RESULT,false,true,gridDataBase64);
                    getGridXMLById_resultHandler(resultEvt);
                }
            }
            
            private function selectItemRenderer(item : Object) : ClassFactory {
                var sq : Square = item as Square;
                var classFactory : ClassFactory;
                
                if (sq.type == SquareType.DEFINITION){
                    classFactory = new ClassFactory(DefinitionItemRenderer);
                }else{
                    classFactory = new ClassFactory(LetterItemRenderer);
                }
                
                return classFactory;
            }

            private function gridUI_keyUpHandler(event:KeyboardEvent):void{
                var selectedSquare : Square = gridUI.selectedItem as Square;
                var num : int = event.keyCode;
                
                if (num > 64 && num < 91){
                    var strCaps:String = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
                    var char : String = strCaps.charAt(num - 65);
                    selectedSquare.letter = char;
                    if(!_inCorrection){
                        correctGrid();
                    }
                    (gridUI.dataProvider as ArrayCollection).itemUpdated(selectedSquare);
                    autoSelectNextCell();
                }else if(event.keyCode == Keyboard.DELETE){
                    selectedSquare.letter = "";
                    (gridUI.dataProvider as ArrayCollection).itemUpdated(selectedSquare);
                }else if(event.keyCode == Keyboard.BACKSPACE){
                    selectedSquare.letter = "";
                    (gridUI.dataProvider as ArrayCollection).itemUpdated(selectedSquare);
                }else if(event.keyCode == Keyboard.CONTROL){
                    if(_grid.typingDirection == TypingDirection.DOWN){
                        _grid.typingDirection = TypingDirection.RIGHT;
                    }else{
                        _grid.typingDirection = TypingDirection.DOWN;
                    }
                    (gridUI.dataProvider as ArrayCollection).itemUpdated(selectedSquare);
                }
            }
            
            private function validateCorrectionButton_clickHandler(event:MouseEvent):void {
                correctGrid();
            }
            
            private function correctGrid() : void {
                var correctionLevel : int = hintCombo.selectedItem.value;
                
                if (correctionLevel == 2){
                    validateLetter();
                }
                if (correctionLevel == 1){
                    validateWords();
                }
                if (isGridFull()){
                    if (isGridValid()){
                        validateGrid();
                        showEndGamePopup(true);
                    }else{
                        showEndGamePopup(false);
                    }
                }
            }
            
            private var _endGamePopup : EndGameWindow;
            private function showEndGamePopup(win : Boolean ) : void {
                _endGamePopup = PopUpManager.createPopUp(this, EndGameWindow, true) as EndGameWindow;
                PopUpManager.centerPopUp(_endGamePopup);
                _endGamePopup.currentState = win ? "win" : "errors";
                _endGamePopup.onCloseFunction = win ? onWinClose : onErrorsClose;
                _endGamePopup.onContinueFunction = onContinueFunction;
            }
            
            private function onWinClose() : void {
                lockGrid();
            }
            
            private function onErrorsClose() : void {
                _grid.showSolution = true;
                validateGrid();
                lockGrid();
            }
            
            private function onContinueFunction() : void {
                _inCorrection = true;
            }
            
            private function lockGrid() : void {
                _inCorrection = false;
                gridUI.selectedIndex = -1;
                gridUI.enabled = false;
                (gridUI.dataProvider as ICollectionView).refresh();
            }
            
            private function isGridValid() : Boolean {
                var allValid : Boolean = true;
                var sq : Square;
                
                for each(sq in _grid.squares){
                    if (sq.type == SquareType.LETTER && sq.letter != sq.solution){
                        allValid = false;
                        break;
                    }
                }
            
                return allValid;
            }
                
            private function validateGrid() : void {
                var sq : Square;
                
                for each(sq in _grid.squares){
                    if (sq.type == SquareType.LETTER){
                        sq.solved = sq.letter == sq.solution;
                    }
                }
            }
            
            private function isGridFull() : Boolean {
                var isFull : Boolean=true;
                var sq : Square;
                
                for each(sq in _grid.squares){
                    if (sq.type == SquareType.LETTER && sq.letter == ""){
                        isFull = false;
                        break;
                    }
                }
                
                return isFull;
            }
            
            
            private function validateLetter(idx : int=-1) : Boolean {
                if (-1 == idx){
                    idx = gridUI.selectedIndex;
                }
                var sq : Square = _grid.squares[idx] as Square;
                
                sq.solved = sq.letter == sq.solution;
                return sq.solved;
            }
            
            private function validateWords(idx : int=-1) : Boolean {
                if (-1 == idx){
                    idx = gridUI.selectedIndex;
                }
                return false;
            }
            
            private var _programmaticalSelection : Boolean=false;
            private function autoSelectNextCell() : void {
                if (gridUI.selectedIndex == -1) return;
                
                _programmaticalSelection = true;
                
                var idx : int = gridUI.selectedIndex;
                var rowIdx : int = Math.floor(idx / _grid.colCount);
                var colIdx : int = idx - (rowIdx * _grid.colCount);
                var nextSelectedIdx : int = idx;
                var sq : Square = _grid.squares[idx];
                
                if(_grid.typingDirection == TypingDirection.RIGHT){
                    do{
                        nextSelectedIdx++;
                        
                        // If we are on the last accessible cell of the grid,
                        // jump to the first cell of the grid.
                        if (nextSelectedIdx >= _grid.squares.length){
                            nextSelectedIdx = 0;
                        }
                        sq = _grid.squares[nextSelectedIdx];
                        
                    }while(sq.type != SquareType.LETTER)
                }else{                    
                    do{
                        nextSelectedIdx += _grid.colCount;
                        
                        // If we are on the last cell of the column,
                        // jump to first cell of the next column, if possible.
                        if (nextSelectedIdx >= _grid.squares.length){
                            nextSelectedIdx = colIdx >= _grid.colCount-1 ? 0 : colIdx + 1;
                        }
                        
                        sq = _grid.squares[nextSelectedIdx];
                        
                    }while(sq.type != SquareType.LETTER)
                }
                gridUI.selectedIndex = nextSelectedIdx;
            }

            private function getBestTypingDirection() : String {
                // Default is RIGHT typing
                var result : String = TypingDirection.RIGHT;
                var sq : Square = gridUI.selectedItem as Square;
                
                if (sq){
                    
                    // If no horizontal definition starts here --> DOWN typing
                    if (sq.arrows.indexOf(ArrowType.HOR) == -1
                        && sq.arrows.indexOf(ArrowType.HOR_TOP) == -1
                        && sq.arrows.indexOf(ArrowType.HOR_UNDER) == -1){
                        result = TypingDirection.DOWN;
                    }
                    
                    // If no vertical definition starts here --> RIGHT typing
                    if (sq.arrows.indexOf(ArrowType.VER) == -1
                        && sq.arrows.indexOf(ArrowType.VER_RIGHT) == -1
                        && sq.arrows.indexOf(ArrowType.VER_RIGHT_TOP) == -1){
                        result = TypingDirection.RIGHT;
                    }
                }
                return result;
            }
            
            private function gridUI_changingHandler(event:IndexChangeEvent):void{
                if(!_programmaticalSelection){
                    _grid.typingDirection = getBestTypingDirection();
                }
                _programmaticalSelection = false;
            }
            
            private function solutionChckBx_changeHandler(event:Event):void {
                _grid.showSolution = solutionChckBx.selected;
                (gridUI.dataProvider as ICollectionView).refresh();
            }

            /* SERVICES */
            
            private function getGridXMLById_resultHandler(event:ResultEvent):void{
                var gridBuilder : GridBuilder = new GridBuilder();
                //var retrievedData : XML = event.result as XML;
                
                var gridDataBase64 : String = event.result as String;
                var decoder : Base64Decoder = new Base64Decoder();
                try{
                    decoder.decode(gridDataBase64)
                    var gridDataBA : ByteArray = decoder.toByteArray();
                    var gridDataStr : String = gridDataBA.toString();
                    
                    var gridDataXML : XML = new XML(gridDataStr);
                    if(!gridDataXML){
                        Alert.show("Bad XML: " + gridDataStr);
                    }
                    if (gridBuilder.parseXML(gridDataXML)){
                        _grid = gridBuilder.parsedGrid;
                        (gridUI.layout as TileLayout).requestedColumnCount = _grid.colCount;
                        (gridUI.layout as TileLayout).requestedRowCount = _grid.rowCount;
                        gridUI.dataProvider = new ArrayCollection(_grid.squares);
                        title.text = _grid.title;
                        subtitle.text = "by " + _grid.author + "(level: " + _grid.level + ")";
                        gridUI.setFocus();
                    }else{
                        Alert.show("Could not build grid from: " + gridDataStr);
                    }
                }catch(e:*){
                    getGridXMLById.send({id : _requestedGridId});
                }
                
            }

            private function getGridXMLById_faultHandler(event:FaultEvent):void{
                Alert.show("Error while retrieving Grid with id" + _requestedGridId);
            }
