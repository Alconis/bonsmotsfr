package model
{
    import mx.events.PropertyChangeEvent;
    import mx.events.PropertyChangeEventKind;

    public class Square
    {
        // Index of the square in the grid.
        public var idx : int;
        
        // Type can be DEFINITION or LETTER
        public var type : String;
        
        // Definitions displayed in the square of type DEFINITION
        public var definitions : Array /* Of Definition */=[];
        
        // Arrows displayed in the square of type LETTER
        public var arrows : Array /* Of ArrowType */ =[];
        
        // Letter typed by the player in this square
        public var letter : String="";
        
        // Reference to the owning grid to ease accesses
        public var gridRef : Grid;
        
        // Letter that the user have to find for this square
        public var solution : String;
        
        // true if letter matches the solution.
        // if true, the user can no longer edit it.
        public var solved : Boolean=false;
    }
}
