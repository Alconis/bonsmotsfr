<?xml version="1.0" encoding="utf-8"?>
<s:ItemRenderer xmlns:fx="http://ns.adobe.com/mxml/2009" 
                xmlns:s="library://ns.adobe.com/flex/spark"
                autoDrawBackground="false"
                depth="0" depth.hovered="1"
                >
    <fx:Script>
        <![CDATA[
            import enums.DefinitionOrientation;
            
            import flashx.textLayout.formats.TextAlign;
            import flashx.textLayout.formats.VerticalAlign;
            import flashx.textLayout.formats.WhiteSpaceCollapse;
            
            import model.Definition;
            import model.Square;
            
            import spark.components.RichText;
            import spark.primitives.Line;
                        
            [Bindable]private var _isFirstCol : Boolean;
            [Bindable]private var _isLastCol : Boolean;
            [Bindable]private var _isFirstRow : Boolean;
            [Bindable]private var _isLastRow : Boolean;
            
            override public function set data(value:Object):void {
                super.data = value;
                
                if (!value) return;
                
                defContainer.removeAllElements();
                
                var sq : Square = value as Square;
                var rowIdx : int = Math.floor(sq.idx / sq.gridRef.colCount);
                var colIdx :int = sq.idx - (rowIdx * sq.gridRef.colCount);
                
                _isFirstCol = colIdx == 0;
                _isLastCol = colIdx == sq.gridRef.colCount - 1;
                _isFirstRow = rowIdx == 0;
                _isLastRow = rowIdx == sq.gridRef.rowCount - 1;
                
                for each(var def : Definition in sq.definitions){
                    var textField : RichText = new RichText();
                    textField.maxDisplayedLines = -1;
                    textField.setStyle("textAlign",TextAlign.CENTER);
                    textField.setStyle("verticalAlign",VerticalAlign.MIDDLE);
                    textField.setStyle("whiteSpaceCollapse", WhiteSpaceCollapse.COLLAPSE);
                    textField.text = def.text;
                    textField.percentWidth = 100;
                    textField.percentHeight = 100 / sq.definitions.length;
                    
                    defContainer.addElement(textField);
                    
                    if (sq.definitions.length > 1 && sq.definitions.indexOf(def) == 0){
                        var divider : Line = new Line();
                        divider.percentWidth = 100;
                        divider.stroke = new SolidColorStroke();
                        defContainer.addElement(divider);
                    }
                }
            }
        ]]>
    </fx:Script>
    
    <s:postLayoutTransformOffsets>
        <s:TransformOffsets id="offsets"
                            x.hovered="{_isFirstCol?0:(_isLastCol?-width:-width/2)}"
                            y.hovered="{_isFirstRow?0:(_isLastRow?-height:-height/2)}"
                            scaleX.hovered="2"
                            scaleY.hovered="2"/>
    </s:postLayoutTransformOffsets>
    
    <s:states> 
        <s:State name="normal"/> 
        <s:State name="hovered"/> 
        <s:State name="selected"/> 
    </s:states> 
    
    <s:Rect id="rect" 
            left="0" right="0" top="0" bottom="0" 
            alpha="1.0"> 
        <s:stroke> 
            <s:SolidColorStroke color="#000000"
                                weight="1"/> 
        </s:stroke>
        <s:fill>
            <s:SolidColor color="#F5F7B5" color.selected="#A0A260"/>
        </s:fill>
    </s:Rect>     
    
    <s:Group id="defContainer" width="100%" height="100%"
             fontSize="8" fontSize.hovered="6">
        <s:layout>
            <s:VerticalLayout paddingTop="2"/>
        </s:layout>
    </s:Group>
</s:ItemRenderer>
